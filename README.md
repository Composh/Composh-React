## Install

```bash
npm install --save composh-react
```

```bash
yarn add composh-react
```



## Usage

```jsx
import React from 'react'

import {
  Accordion,
} from 'composh-react'



const ScreenName: React.FC = () => {
  return (

    <Accordion />

  )
}
```
